<dependency>

  <artifactId>org-java</artifactId>

</dependency>
#+END_SRC

#+BEGIN_SRC groovy
  compile 'com.orgzly:org-java:1.2.2'
#+END_SRC

** SBT

#+BEGIN_SRC scala
  libraryDependencies += "com.orgzly" % "org-java" % "1.2.2"
#+END_SRC
